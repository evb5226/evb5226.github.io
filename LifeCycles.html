// Life Cycles
  // By Emily Burke
  // The loops of life and a series of functions
  // This code can be used in conjunction with a mechatronic set up
  // The mechatronic setup involves an LED (with wires and a resistor).

//to run once on start
void setup(){
  pinMode(13, OUTPUT);                  //this is all the user will see through the LED
}
                                        //an infinite loop which creates LIFE cycles
void loop() {
  l();
  i();
  f();
  e();
  renew();
}
                                        //code for L (.-..)
void l (){
  ShortBeep();
  LongBeep();
  ShortBeep();
  ShortBeep();
  CharacterPause();
}
                                        //code for I (..)
void i(){
  ShortBeep();
  ShortBeep();
  CharacterPause();
}
                                        //code for F (..-.)
void f(){
  ShortBeep();
  ShortBeep();
  LongBeep();
  ShortBeep();
  CharacterPause();
}
                                        //code for E (.)
void e(){
  ShortBeep();
}
                                        //code for Dot
void ShortBeep() {
  digitalWrite(13, HIGH);
  dotLength();
  digitalWrite(13, LOW);
  ElementPause();
  }
                                        //code for Dash
void LongBeep(){
  digitalWrite(13, HIGH);
  dashLength();
  digitalWrite(13, LOW);
  ElementPause();
  }

void ElementPause(){
  delay(500);
}
void CharacterPause(){
  delay(1500);
}
void renew(){
  delay(3500);
}
void dotLength(){
  delay(500);
}
void dashLength(){
  delay(1500);
}
