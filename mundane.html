//Global Variables
  //Characters
  int Engineer=3;
  int CarSalesman=4;

  //Life Changers
  int God=1;
  int Fate=2;

  //Status of Characters
  int EngineDesigner=6;
  int Salesman =7;

  //Parts of Life
  int Number;
  int Day;
  int DailyLifeEngineer;
  int DailyLifeCarSalesman;

  bool Life=HIGH;

//Run Once at Start
void setup() {
  pinMode(Engineer, INPUT);                   //Engineer comes as an input with control
  pinMode(CarSalesman, INPUT);                //Salesman also comes as an input
  pinMode(EngineDesigner, OUTPUT);            //The life of the engineer in reality is an output
  pinMode(Salesman, OUTPUT);                  //The salesman's life is also just an output  
  attachInterrupt(God, Good, RISING);         //God can interrupt for better
  attachInterrupt(God, Death, FALLING);       //God can interrupt for worse
  attachInterrupt(Fate, Guessing, CHANGE);    //Leaving things to fate is chance
  Serial.begin(9600);                         //People are always watching
}

//To Run Repeatedly Unless Interrupted
void loop() {
  do{                                                   //Tasks to do while alive
    for (1; Day<365; Day=Day+1){                        //Every day in the year
    DailyLifeEngineer=digitalRead(Engineer);            //Daily life is assigned by reading pin
    DailyLifeCarSalesman=digitalRead(CarSalesman);      //Daily life is assigned by reading pin
    }
  }while (Life==HIGH);                                  //Do for the rest of life
}

//When God interrupts for Good
void Good(){              
  digitalWrite(EngineDesigner,HIGH);                    //Engineer feels good
  digitalWrite(Salesman,HIGH);                          //Car Salesman feels good
}

//When Fate interrupts
void Guessing(){
  Number=random(2);                                 //Random number generation
  if (Number==0){                                   //If random number 
    digitalWrite(EngineDesigner, LOW);              //Both feel low
    digitalWrite(Salesman, LOW);
  }
  else if (Number==1){                              //if random number
    digitalWrite(EngineDesigner, HIGH);             //Both feel high
    digitalWrite(Salesman, HIGH);
  }
  else{                                             //if random number
    digitalWrite(EngineDesigner,random(1));         //Engineer may feel good or bad
    digitalWrite(Salesman,random(1));               //CarSalesman may feel good or bad
  } 
}

//When God interrupts for Bad
void Death(){
  Life=LOW;             //All those die
}
